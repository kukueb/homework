#Принимаем значения катетов прямоугольного треугольника
#приписка int перед input означает, что переменную нужно воспринимать как целое число, типа integer
a = int(input('длина первого катета равна'))
b = int(input('длина второго катета равна'))
# ** - возведение в квадрат. 
print('гипотенуза равна' , (a**2+b**2)**0.5)

c = int(input('Первый коэффициент равен: '))
d = int(input('Второй коэффициент равен: '))
e = int(input('Третий коэффициент равен: '))

disc = (d**2 - 4*c*e) / 4 # Считаем дискриминант по формуле
print('Дискриминант равен:', disc) #Вывод дискриминанта
