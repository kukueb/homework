temp = int(input('Сколько у вас грасов за окном? '))
if temp >= 30:
    print('У вас там очень жарко! Советуем залезть в холодильник')
elif temp >= 25:
    print('Жарковато, советуем примотать к себе скотчем винтелятор')
elif temp >= 15:
    print('Тепло, внатуре тепло, просто тепло, чего уж там...')
elif temp >= 5:
    print('Прохладно, ладно, шапку не забудь, а то мозги простудишь...')
elif temp >= -15:
    print('Холодно, шапка не поможет...')
else:
    print('Очень холодно, берём тяжёлую артиллерию - подштанники')
